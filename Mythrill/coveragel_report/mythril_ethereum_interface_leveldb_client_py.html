19# Per https://github.com/ethereum/go-ethereum/blob/master/core/rawdb/schema.go 

20# prefixes and suffixes for keys in geth 

21header_prefix = b"h" # header_prefix + num (uint64 big endian) + hash -> header 

22body_prefix = b"b" # body_prefix + num (uint64 big endian) + hash -> block body 

23num_suffix = b"n" # header_prefix + num (uint64 big endian) + num_suffix -> hash 

24block_hash_prefix = b"H" # block_hash_prefix + hash -> num (uint64 big endian) 

25block_receipts_prefix = ( 

26 b"r" # block_receipts_prefix + num (uint64 big endian) + hash -> block receipts 

27) 

28# known geth keys 

29head_header_key = b"LastBlock" # head (latest) header hash 

30# custom prefixes 

31address_prefix = b"AM" # address_prefix + hash -> address 

32# custom keys 

33address_mapping_head_key = b"accountMapping" # head (latest) number of indexed block
